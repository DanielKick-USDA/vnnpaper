def _dist_scale_function(out, dist, decay_rate):
    scale = 1/(1+decay_rate*dist)
    out = round(scale * out)
    out = max(1, out)
    return out

def _expand_node_shortcut(vnn_helper = myvnn, query = 'y_hat'):
    # define new entries
    if True in [True if e in vnn_helper.edge_dict.keys() else False for e in 
                [f'{query}_res_-2', f'{query}_res_-1']
                ]:
        print('Warning! New node name already exists! Overwriting existing node!')

    # Add residual connection in graph
    vnn_helper.edge_dict[f'{query}_res_-2'] = myvnn.edge_dict[query] 
    vnn_helper.edge_dict[f'{query}_res_-1'] = [f'{query}_res_-2']
    vnn_helper.edge_dict[query]             = [f'{query}_res_-2', f'{query}_res_-1']

    # Add new nodes, copying information from query node
    vnn_helper.node_props[f'{query}_res_-2'] = vnn_helper.node_props[query] 
    vnn_helper.node_props[f'{query}_res_-1'] = vnn_helper.node_props[query]

    return vnn_helper

To constrain the number of parameters to optimize over this model will follow these steps:

  1. Initialize the graphs structure
  2. Initialize default output values for each node type (input, edge, output)
  3. Initialize default dropout values for each node type (input, edge, output)
  4. Apply scaling with respect to a nodes distance from the output node
    1. This factor should have a value which does not apply any scaling and on that applys severe scaling
    2. Depending on the function used, it may make sense to transform this value e.g. use log10(S) instead of S
    3. An option for this would be to use an inverse of a root: \(scale = \frac{1}{1+ rate*\sqrt{dist}}\)
    4. A simpler option would be: \(scale = \frac{1}{1+ rate*dist}\)
    5. Using 0 or 1 based indexing will be relevant here.
  5. Expand the graph to replicate any nodes that should be replicated.
    1. NOTE: The order here is key. Applying scaling before expanding the graph will result in a smaller scaling penalty than if it is expanded before.
  6. Insert shortcuts, likely for nodes at certain levels or for edge nodes.

class NeuralNetwork(nn.Module):
    def __init__(self, layer_list):
        super(NeuralNetwork, self).__init__()
        self.layer_list = nn.ModuleList(layer_list)
 
    def forward(self, x):
        for l in self.layer_list:
            x = l(x)
        return x

    # I'll use the lowest average error in an epoch
    log_path = lightning_log_dir+'/'+exp_name
    fls = os.listdir(log_path)
    nums = [int(e.split('_')[-1]) for e in fls] 

    M = pd.read_csv(log_path+f"/version_{max(nums)}/metrics.csv")
    M = M.loc[:, ['epoch', 'train_loss']].dropna()

    M = M.groupby('epoch').agg(
        train_loss = ('train_loss', 'mean'),
        train_loss_sd = ('train_loss', 'std'),
        ).reset_index()

    train_metric = M.train_loss.min()
    print(train_metric)
    return {"train_loss": (train_metric, 0.0)}
